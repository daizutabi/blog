GCC vs CL

Go言語からCUDAを使うために，簡単な実験を行います．

まずはCUDAのcuInit関数を呼ぶだけの簡単なCソースをコンパイルします．

// init.c
#include <stdio.h>
#include <cuda.h>

int main() {
    printf("cuInit: %d", cuInit(0));
}

1 C - MinGW-w64

> set CPATH=C:\cuda\v9.1\include
> set LIBRARY_PATH=C:\cuda\v9.1\lib\x64
> gcc init.c -lcuda
> a.exe
cuInit: 0

次にGoからCUDAを呼び出します．

// main.go
package main

//#include <stdio.h>
//#include <cuda.h>
import "C"
import "fmt"

func main() {
    fmt.Println("cuInit: ", C.cuInit(C.uint(0)))
}

4 Go - MinGW-w64

> set CGO_CFLAGS=-IC:\cuda\v9.1\include
> set CGO_LDFLAGS=-LC:\cuda\v9.1\lib\x64 -lcuda
> go run main.go
cuInit: 0
